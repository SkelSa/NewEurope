ENG_historical_military_operations = { 

	ENG_the_peking_plan = {
	
		icon = eng_poland_flag

		allowed = {
			tag = ENG
		}

		fire_only_once = yes

		available = {
			POL = {
				OR = { 
					is_in_faction_with = ENG
					ENG = { has_guaranteed = POL }
					has_opinion = { target = ENG value > 10 }
				}
			}
			GER = { focus_progress = { focus = GER_danzig_or_war progress > 0.85 } }
		}

		cost = 15

		ai_will_do = {
			base = 45
			modifier = {
				factor = 200
				is_historical_focus_on = yes
			}
		}
		
		visible = {
			GER = {
				focus_progress = { focus = GER_danzig_or_war progress > 0.01 }
				OR = {
					NOT = { is_in_faction_with = POL }
					NOT = { is_in_faction_with = ENG }
				}
			}
			POL = {
				OR = { 
					has_government = democratic
					has_government = neutrality
				}
				NOT = {
					OR = { 
						is_in_faction_with = GER
						has_war_with = GER
					}
				}
			}
			any_country = { 
				has_government = democratic
				NOT = { has_war_with = POL }
			}
		}
		
		complete_effect = {
			POL = { country_event = { days = 1 id = ne_misc.22 } }
		}
	}
}

ENG_historical_missions_ne = {

    ENG_the_english_poland_alliance = {
	
		icon = eng_poland_flag

		allowed = { 
			tag = ENG
		}         
	
		available = { ##conditions of implementation
			OR = { 
				POL = { is_in_faction_with = ENG }
				AND = {
					ENG = { has_guaranteed = POL }
					POL = { has_opinion = { target = ENG value > 50 } }
				}
			}
		}
	
		fire_only_once = yes

		ai_will_do = {
			base = 85
			modifier = {
				factor = 200
				is_historical_focus_on = yes
			}
		}

		activation = {
			GER = {
				has_completed_focus = GER_end_of_czechoslovakia
				OR = {
					NOT = { is_in_faction_with = POL }
					NOT = { is_in_faction_with = ENG }
				}
			}
			POL = {
				OR = { 
					has_government = democratic
					has_government = neutrality
				}
				NOT = { 
					is_in_faction_with = GER
					OR = {
						ENG = { has_guaranteed = POL }
						ENG = { is_in_faction_with = POL }
						AND = {
							POL = { has_opinion = { target = ENG value > 40 } }
							ENG = { is_ai = no }
						}
					}
				}
			}
			any_country = { 
				has_government = democratic
				NOT = { has_war_with = POL }
			}
		}
	
		is_good = no                        
	
		days_mission_timeout = 45     
	
		timeout_effect = {                     
			POL = {
				add_opinion_modifier = { target = ENG modifier = allies_didnt_guarantee } 
			}
			ENG = {
				add_political_power = -150
			}
		}
	
		complete_effect = {                    
			if = {
				limit = {
					OR = {
						POL = { is_in_faction_with = ENG }
						has_global_flag = anglo_guarantee
					}
				}
				ENG = { add_political_power = 45 }
			}
			if = {
				limit = {
					NOT = { has_global_flag = anglo_guarantee }
				}
				ENG = {
					add_political_power = 25
					hidden_effect = { set_global_flag = anglo_guarantee }
					custom_effect_tooltip = anglo_guarantee_tooltip.1.tt
				}
			}
		}

    }

}
