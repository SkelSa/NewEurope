ne_iran_war_preparation = {

	enable = {
		OR = {
			tag = SOV
			tag = RAJ
			tag = ENG
		}
		AND = {
			SOV = { has_war_with = GER }
			ENG = { has_war_with = GER }
			NOT = {
				OR = {
					SOV = { has_war_with = ENG }
					ENG = { has_war_with = SOV }
				}
			}
		}

		PER = {
			exists = yes
			NOT = {
				OR = { 
					has_government = communism
					has_government = democratic
				}
				OR = { 
					is_in_faction_with = SOV
					is_in_faction_with = ENG
				}
				OR = {
					has_war_with = SOV
					has_war_with = ENG
				}
			}
		}
	}
	
	abort = {
		PER = {
			AND = {
				has_war_with = SOV
				has_war_with = ENG
			}
			OR = {
				surrender_progress > 0.024
				has_capitulated = yes
			}
		}
	}
	
	ai_strategy = {
		type = conquer
		id = "PER"
		value = 80
	}

	ai_strategy = {
		type = antagonize
		id = "PER"
		value = 80
	}
}

ne_eng_pol_guarantee = {

	enable = {
		tag = ENG
		GER = {
			has_completed_focus = GER_end_of_czechoslovakia
			OR = {
				NOT = { is_in_faction_with = POL }
				NOT = { is_in_faction_with = ENG }
			}
		}

		POL = {
			OR = { 
				has_government = democratic
				has_government = neutrality
			}
		}

		NOT = { 
			OR = {
				ENG = { has_guaranteed = POL }
				ENG = { is_in_faction_with = POL }
				POL = { is_in_faction_with = GER }
				AND = {
					POL = { has_opinion = { target = ENG value > 40 } }
					ENG = { is_ai = no }
				}
			}
		}

		any_country = { 
			has_government = democratic
			NOT = { has_war_with = POL }
		}
	}
	
	abort = {
		POL = { has_war_with = GER }
	}
	
	ai_strategy = {
		type = protect ##in this case means that english will guarantee independence poland
		id = "POL"
		value = 200
	}

}
