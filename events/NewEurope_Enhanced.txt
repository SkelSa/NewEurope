		is_ai = no

				has_war_with = FRA

			}

		}

			SOZ = { exists = yes } 
			FOZ = { exists = yes } 
			AOZ = { exists = yes } 
			BOZ = { exists = yes }
			GER = { exists = no }
			
			AND = {		##GER is puppet ( usually of SOV )		
				GER = {
					exists = yes
					NOT = { has_government = fascism }
					is_subject = yes
				}
			}
			
			AND = {		##GER is liberated, not played by player		
				GER = {
					exists = yes
					has_government = democratic
					is_ai = yes
				}
			}
			
		}
	}
	
	mean_time_to_happen = {	days = 40 }
	
	option = {
		name = ne_intro.1.a        #start
		hidden_effect = {
			country_event = ne.1
		}
	}
	option = {
		name = ne_intro.1.b         #close		
	}
	
}
