	picture = GFX_somepic

	fire_only_once = yes

	
	trigger = {

		tag = FIN
		
		FIN = {
			exists = yes
			is_subject = no
			NOT = { has_war_with = SOV }
		}
		
		GER = {
			exists = yes
			is_faction_leader = yes 
			has_government = fascism
		}
		
		NOT = {		
			FIN = { controls_state = 146 } 
			FIN = { owns_state = 146 } 		
		}
		
		GER = { has_war_with = SOV}
		SOV = { surrender_progress > 0.03 }
		
	
	}
		
	mean_time_to_happen = {	days = 6 }

	
	option = { # join the war
		name = ne_finland.1.a
		ai_chance = {
			factor = 100
		
			modifier = {
				factor = 0
					OR = {
						has_government = communism
						has_war = yes
					}
				 			
			}			
		}

		hidden_effect = {
			country_event = ne_finland.2			
		}				
		
	}
	
	
	option = { # stay out of it, war is hell
		name = ne_finland.1.b
		ai_chance = {
			factor = 0
		
			modifier = {
				factor = 100
				OR = {
					has_government = communism
					has_war = yes
				}
			}			
		}
		
		add_political_power = -30
		add_national_unity = -0.05
		add_named_threat = { threat = -5 name = threat_ne_finland_neutrality }
		
	}
	
	
}
