					owns_state = 137

			}
	}
	
	mean_time_to_happen = { days = 265 }

	option = {
		name = ne_ssrwb.1.a
		ai_chance = { factor = 100 }

				limit = {
					UKR = {
						AND = {
							exists = yes
							is_puppet_of = SOV
						}	
					}
				}
				UKR = {
					transfer_state = 137
					add_opinion_modifier = { target = SOV modifier = huge_increase }
				}
				137 = { 
						remove_core_of = SOV
						add_core_of = UKR 
						add_claim_by = UKR
					}
			}
		if = {
				limit = {
					UKR = {
						NOT = { exists = yes }
					}
				}
				137 = { 
						remove_core_of = SOV
						add_core_of = UKR 
						add_claim_by = UKR
					}
		137 = {
			add_building_construction = {
				type = infrastructure
				level = 2
				instant_build = yes
			}
			add_building_construction = {
				type = naval_base
				level = 2
				province = 9680
				instant_build = yes
			}
			add_building_construction = {
				type = infrastructure
				level = 4
				instant_build = no
			}
			add_building_construction = {
				type = naval_base
				level = 4
				province = 9680
				instant_build = no
			}
