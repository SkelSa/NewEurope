		}

	}

			factor = 75
			
		}
		
		if = {
			limit = { 
				has_government = democratic			
			}
			
			USA = { country_event = { days = 3 id = ne_faction.4 } }		
		}
		
		if = {
			limit = { 
				has_government = communism			
			}
			
			SOV = { country_event = { days = 3 id = ne_faction.4 } }		
		}
	
	}

			factor = 25			
		}

}

				}
			}

		}

				TAN = {
				
					AND = {

						is_subject_of = SOV
					}
				}

			}

				ROM = {				
					OR = {

					}

			}
			
		ROM = { country_event = { days = 4 id = ne_wp.2 } }

				OR = {

			}

				OR = {
